使用 <chrono> 库#

<chrono> 是 C++ 标准库中的时间库，精度高且无需依赖平台特定 API，适合大多数场景。

核心步骤：

  1. 用 std::chrono::high_resolution_clock（高精度时钟）记录函数执行前的时间点 start；
  2. 调用需要测试的函数；
  3. 记录函数执行后的时间点 end ；
  4. 计算 end - start 的时间差，转换为毫秒（ms）、微妙（us）等直观单位。
#include <iostream>
#include <vector>
#include <chrono>

// 待测试的函数：查找 vector 中的最大值
int findMax(const std::vector<int>& scores) {
    int maxScore = scores[0];
    for (int score : scores) {
        if (score > maxScore) maxScore = score;
    }
    return maxScore;
}


int main() {
    // 1. 准备测试数据（例如 100 万个随机成绩）
    const int dataSize = 1000000;
    std::vector<int> scores(dataSize);
    for (int i = 0; i < dataSize; ++i) {
        scores[i] = rand() % 100; // 生成 0-99 的随机成绩
    }
    
    // 2. 记录开始时间（高精度时钟）
    auto start = std::chrono::high_resolution_clock::now();
    
    // 3. 调用待测试函数
    int result = findMax(scores);
    
    // 4. 记录结束时间
    auto end = std::chrono::high_resolution_clock::now();
    
    // 5. 计算时间差（转换为毫秒）
    auto duration = std::chrono::duration_cast<std::chrono::milliseconds>(end - start);
    
    std::cout << "函数 findMax 执行时间：" << duration.count() << "毫秒\n";
    std::cout << "找到的最大值：" << result << std::endl;
    
    return 0;
}
